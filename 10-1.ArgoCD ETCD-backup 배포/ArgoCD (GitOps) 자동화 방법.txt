ArgoCD (GitOps) 자동화 방법

# Sealed Secrets Helm 저장소 추가 및 설치
1. helm repo add sealed-secrets https://bitnami-labs.github.io/sealed-secrets
2. helm repo update
3. helm install sealed-secrets-controller sealed-secrets/sealed-secrets --namespace kube-system

# kubeseal(리눅스용) 설치
1. wget https://github.com/bitnami-labs/sealed-secrets/releases/download/v0.26.2/kubeseal-0.26.2-linux-amd64.tar.gz
2. tar -xvzf kubeseal-0.26.2-linux-amd64.tar.gz kubeseal
3. mv kubeseal /usr/local/bin/kubeseal

# 암호화 명령어 실행 및 확인
1. kubectl create secret generic etcd-webhook-secret   --namespace monitoring   --from-literal=WEBHOOK_URL="https://hooks.slack.com/services/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"   --dry-run=client -o yaml | kubeseal --format=yaml > sealed-secret.yaml
2. cat sealed-secret.yaml
3. sealed-secret.yaml의 WEBHOOK_URL에 암호화된 값 입력 후 github 업로드

# 차트 저장소 추가 및 네임스페이스 생성

# ArgoCD Applications 설정

1. General (기본 설정)

Application Name: etcd-backup
Project Name: default
SYNC POLICY: Automatic (체크 시 GitHub에 코드가 올라가면 즉시 K8s에 자동 반영됩니다.)
세부 옵션: Prune Resources(Git에서 파일 지우면 K8s에서도 삭제)와 Self Heal(누가 K8s에서 강제로 수정하면 Git 기준으로 원복)도 켜주시면 완벽한 GitOps가 됩니다.

2. Source (GitHub 저장소 설정)

Repository URL: GitHub 레포지토리 주소 (https://github.com/gibeom01/IDC_Physical_Server_K8s.git)
Revision: HEAD 또는 main (브랜치 이름)
Path: k8s-infra/backup (GitHub 레포지토리 내에서 sealed-secret.yaml과 cronjob.yaml이 들어있는 폴더 경로)

3. Destination (배포될 K8s 클러스터 위치)

Cluster URL: [https://kubernetes.default.svc](https://kubernetes.default.svc) (현재 ArgoCD가 설치된 내부 클러스터)
Namespace: monitoring (이전에 SealedSecret을 만들 때 지정했던 네임스페이스)

# 해독된 일반 Secret 생성 확인

kubectl get secret etcd-webhook-secret -n monitoring
kubectl get secret etcd-webhook-secret -n monitoring -o yaml (Base64로 인코딩된 상태 출력되면 정상)

# 백업 즉시 실행 및 Slack 알람 테스트

kubectl create job --from=cronjob/etcd-backup-job test-backup-1 -n monitoring (CronJob 기반 1회용 Job 생성)
kubectl get pods -n monitoring | grep test-backup (Pod 상태 ContainerCreating -> Running -> Completed 확인)
파드가 Completed 되는 순간, 사내 메신저(Slack)로 "✅ [node1] etcd 자동 백업 성공" 메시지 도착 확인

# 에러 발생 시
kubectl logs test-backup-1-xxxxx -n monitoring

자주 발생하는 에러
에러 1: K8s CronJob 컨테이너 안 etcd-smart-backup.sh 스크립트 파일이 없는 경우 (스크립트 K8s ConfigMap 만들어서 파드 마운트(Volume) 해주어야 K8s 내부에서 실행)
에러 2: 파드가 마스터 노드 /data/etcd-backup 폴더나 /etc/ssl/etcd/ssl 인증서 폴더에 접근하지 못하는 경우 (HostPath 볼륨 마운트 필요)

# cronjob.yaml

apiVersion: v1
kind: ConfigMap
metadata:
  name: etcd-backup-script
  namespace: monitoring
data:
  etcd-backup.sh: |
    #!/bin/bash
    export ETCDCTL_API=3
    DATE=$(date +%Y%m%d_%H%M)
    BACKUP_DIR="/data/etcd-backup"
    BACKUP_FILE="${BACKUP_DIR}/etcd-${DATE}.db"
    
    ETCDCTL_CMD="etcdctl --endpoints=https://127.0.0.1:2379 --cacert=/etc/ssl/etcd/ssl/ca.pem --cert=/etc/ssl/etcd/ssl/admin-node1.pem --key=/etc/ssl/etcd/ssl/admin-node1-key.pem"
    
    echo "etcd 백업 시작..."
    $ETCDCTL_CMD snapshot save ${BACKUP_FILE} > /dev/null 2>&1
    
    if [ $? -eq 0 ]; then
      $ETCDCTL_CMD --write-out=simple snapshot status ${BACKUP_FILE} > /dev/null 2>&1
      if [ $? -eq 0 ]; then
        echo "백업 성공: ${BACKUP_FILE}"
        curl -s -X POST -H 'Content-type: application/json' --data '{"text":"[node1] etcd GitOps 자동 백업 성공!"}' ${WEBHOOK_URL}
      else
        curl -s -X POST -H 'Content-type: application/json' --data '{"text":"[node1] etcd 백업 파일 손상!"}' ${WEBHOOK_URL}
      fi
    else
      curl -s -X POST -H 'Content-type: application/json' --data '{"text":"[node1] etcd 백업 실패!"}' ${WEBHOOK_URL}
    fi
    
    # 14일 지난 백업 자동 삭제
    find ${BACKUP_DIR} -name "etcd-*.db" -type f -mtime +14 -delete
    echo "오래된 백업 정리 완료."
---
apiVersion: batch/v1
kind: CronJob
metadata:
  name: etcd-backup-job
  namespace: monitoring
spec:
  schedule: "0 3 * * *" # 매일 새벽 3시
  concurrencyPolicy: Forbid
  jobTemplate:
    spec:
      template:
        spec:
          hostNetwork: true #  127.0.0.1(호스트 etcd) 접근 허용
          nodeSelector:
            node-role.kubernetes.io/control-plane: "" # 무조건 마스터 노드에서만 실행
          tolerations:
            - key: node-role.kubernetes.io/control-plane
              operator: Exists
              effect: NoSchedule
          containers:
          - name: backup-container
            image: ubuntu:22.04 # etcdctl이 포함된 이미지 사용
            command: 
                - /bin/bash
                - -c
                - |
                  apt-get update -y && apt-get install -y etcd-client curl
                  /bin/bash /scripts/etcd-backup.sh
            securityContext:
                runAsUser: 0 # root 권한 (인증서 읽기 및 백업 폴더 쓰기 위함)
            env:
              - name: WEBHOOK_URL   # 스크립트의 $WEBHOOK_URL 변수에 값이 들어갑니다.
                valueFrom:
                  secretKeyRef:
                    name: etcd-backup-secret # 복호화된 일반 Secret 이름
                    key: WEBHOOK_URL
            volumeMounts:
                - name: script-volume
                  mountPath: /scripts
                - name: etcd-certs
                  mountPath: /etc/ssl/etcd/ssl
                  readOnly: true
                - name: backup-dir
                  mountPath: /data/etcd-backup
          restartPolicy: OnFailure
          volumes:
            - name: script-volume
              configMap:
                name: etcd-backup-script
            - name: etcd-certs
              hostPath:
                path: /etc/ssl/etcd/ssl
                type: Directory
            - name: backup-dir
              hostPath:
                path: /data/etcd-backup
                type: DirectoryOrCreate
