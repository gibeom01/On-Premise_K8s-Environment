# 테스트용 이미지를 사용해 파드를 4개 생성
kubectl create deployment lb-test --image=ealen/echo-server --replicas=4

# 파드들이 노드 1과 노드 2에 골고루 잘 퍼졌는지 확인
kubectl get pods -l app=lb-test -o wide

# 로드밸런서 타입으로 포트 80번을 개방합니다.
kubectl expose deployment lb-test --type=LoadBalancer --port=80 --target-port=80

# 할당된 로드밸런서 IP(EXTERNAL-IP) 확인
kubectl get svc lb-test

★시나리오 1: 트래픽 분산(부하) 테스트

# 실시간 트래픽 분산 눈으로 보기
while true; do curl -s http://221.139.104.182:32384 | grep "HOSTNAME"; sleep 0.5; done

# 트래픽 폭탄 던지기 (K8s 안에서 부하 발생기(busybox)를 일회성으로 띄워서 요청을 마구 보냅)
kubectl run load-generator --image=busybox --rm -it -- /bin/sh -c "while true; do wget -q -O- http://lb-test; done"

# 노드 1개 강제 제외 (worker에서 진행)

# 파드 상태 확인 (worker에서 진행)
