# iSCSI와 NFS 클라이언트 설치 (모든 노드 공통)
1. sudo apt-get update && sudo apt-get install -y open-iscsi nfs-common
2. sudo systemctl enable --now iscsid

# 충돌 방지 (Multipathd 비활성화, 모든 노드 공통)
1. sudo systemctl stop multipathd
2. sudo systemctl disable multipathd

# 스왑 메모리 비활성화 (모든 노드 공통)
1. sudo swapoff -a

# Helm Longhorn 설치 (Master Node)
1. helm repo add longhorn https://charts.longhorn.io
2. helm repo update
3. helm install longhorn longhorn/longhorn --namespace longhorn-system --create-namespace
4. kubectl get pods -n longhorn-system -w

# MetalLB IPAddressPool 등록
1. cat > metallb-config.yaml

apiVersion: metallb.io/v1beta1
kind: IPAddressPool
metadata:
  name: default-pool
  namespace: metallb-system
spec:
  addresses:
  # 현재 서버와 같은 대역이면서, 다른 기기가 안 쓰고 있는 남는 IP 범위
  - 221.139.104.187-221.139.104.187
---
apiVersion: metallb.io/v1beta1
kind: L2Advertisement
metadata:
  name: default-l2
  namespace: metallb-system
spec:
  ipAddressPools:
  - default-pool

2. kubectl apply -f metallb-config.yaml
3. kubectl rollout restart deployment controller -n metallb-system (MetalLB 캐시 비우기)

# Longhorn UI Volume Test
1. cat > longhorn-test.yaml

apiVersion: v1
kind: PersistentVolumeClaim
metadata:
  name: longhorn-pvc
spec:
  # ★ K8s에게 "이제부터 디스크 Longhorn한테 달라고" 지시
  storageClassName: longhorn
  accessModes:
    - ReadWriteOnce
  resources:
    requests:
      storage: 2Gi
---
apiVersion: v1
kind: Pod
metadata:
  name: volume-test-pod
spec:
  containers:
  - name: ubuntu
    image: ubuntu:22.04
    command: [ "sleep", "3600" ]
    volumeMounts:
    - name: my-volume
      mountPath: /data # 컨테이너 안의 /data 폴더 디스크 연결
  volumes:
  - name: my-volume
    persistentVolumeClaim:
      claimName: longhorn-pvc
---
kind: StorageClass
apiVersion: storage.k8s.io/v1
metadata:
  name: longhorn-static
provisioner: driver.longhorn.io
parameters:
  numberOfReplicas: "2" # 노드 개수 맞게 설정
  staleReplicaTimeout: "2880"

2. kubectl apply -f longhorn-test.yaml

# Longhorn -> Loadblanceer 전환 및 공유 IP 설정
1. kubectl patch svc longhorn-frontend -n longhorn-system -p '{
  "metadata": {
    "annotations": {
      "metallb.universe.tf/allow-shared-ip": "noblesys-shared-ip"
    }
  },
  "spec": {
    "type": "LoadBalancer"
  }
}'
2. kubectl get svc longhorn-frontend -n longhorn-system (IP 할당 확인)

# Toubleshooting

문제 1: Degraded (빨간불) -> 원인: 복제본 개수 불일치 -> 해결: 노드 개수에 알맞게 Update Replicas Count 수정.
문제 2: Scheduling Failure -> 원인: 가용 노드 없음 -> 해결: kubectl uncordon [노드명]으로 스케줄링 허용.
문제 3: Node NotReady -> 원인: 네트워크 충돌 / Kubelet 중지 -> 해결: systemctl restart kubelet, containerd, ARP 캐시 초기화.
문제 4: UI 접속 불가/Pending -> 원인: IP 충돌 (VIP 등) -> 해결: 사용 중인 IP 제외 후 MetalLB Pool 재설정.

# 문제 2의 ARP 캐시 초기화 방법 (문제의 노드에서 진행)

1. sudo ip neigh del 221.139.104.184 dev ens160 (184번 VIP에 대한 잘못된 캐시 정보 삭제)
2. sudo ip neigh flush all (모든 이웃 노드 정보 싹 비우기)
3. curl -k https://221.139.104.183:6443 (자기 자신의 API 서버에 직접 접속 시도)
4. curl -k https://221.139.104.182:6443 (옆 동네의 API 서버에 직접 접속 시도)
5. ip link show | grep mtu (MTU 값 확인 -> 각 노드 실행해 숫자 서로 다른지 확인)
6. sudo reboot (재부팅)
7. watch -n 1 "kubectl get nodes" (노드 깨어나자마자 API 서버 신호 보내는지 실시간 감지)
